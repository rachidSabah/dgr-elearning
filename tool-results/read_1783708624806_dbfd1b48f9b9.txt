   202→function valueToString(v: unknown): string {
   203→  if (v === null || v === undefined) return "";
   204→  if (typeof v === "string") return v;
   205→  if (typeof v === "boolean") return v ? "true" : "false";
   206→  return String(v);
   207→}

   209→export default function SettingsAdminPage() {
   210→  const [activeSection, setActiveSection] = useState<SectionKey>("BRANDING");
   211→  const [values, setValues] = useState<FormState>({ ...DEFAULTS });
   212→  const [loading, setLoading] = useState(true);
   213→  const [saving, setSaving] = useState(false);
   214→  const [dirty, setDirty] = useState<Record<SectionKey, boolean>>({
   215→    BRANDING: false,
   216→    AI: false,
   217→    EMAIL: false,
   218→    INTEGRATION: false,
   219→    GENERAL: false,
   220→  });
   221→
   222→  const fetchSettings = useCallback(async () => {
   223→    setLoading(true);
   224→    try {
   225→      const res = await fetch("/api/admin/settings");
   226→      if (!res.ok) throw new Error("Failed to fetch settings");
   227→      const json: SettingsResponse = await res.json();
   228→      const next: FormState = { ...DEFAULTS };
   229→      for (const [key, entry] of Object.entries(json.data || {})) {
   230→        next[key] = valueToString(entry.value);
   231→      }
   232→      setValues(next);
   233→      setDirty({
   234→        BRANDING: false,
   235→        AI: false,
   236→        EMAIL: false,
   237→        INTEGRATION: false,
   238→        GENERAL: false,
   239→      });
   240→    } catch (err) {
   241→      toast.error("Failed to load settings", {
   242→        description: err instanceof Error ? err.message : "Unknown error",
   243→      });
   244→    } finally {
   245→      setLoading(false);
   246→    }
   247→  }, []);
   248→
   249→  useEffect(() => {
   250→    fetchSettings();
   251→  }, [fetchSettings]);
   252→
   253→  const updateField = (key: string, val: string) => {
   254→    setValues((prev) => ({ ...prev, [key]: val }));
   255→    // Mark the relevant section dirty
   256→    for (const [sectionKey, keys] of Object.entries(SECTION_KEYS)) {
   257→      if (keys.includes(key)) {
   258→        setDirty((prev) => ({ ...prev, [sectionKey]: true }));
   259→        break;
   260→      }
   261→    }
   262→  };
   263→
   264→  const updateBool = (key: string, val: boolean) => {
   265→    updateField(key, val ? "true" : "false");
   266→  };
   267→
   268→  const handleSaveSection = async (section: SectionKey) => {
   269→    const keys = SECTION_KEYS[section];
   270→    const settingsObj: Record<string, unknown> = {};
   271→    for (const k of keys) {
   272→      const raw = values[k] ?? "";
   273→      if (BOOL_KEYS.has(k)) {
   274→        settingsObj[k] = raw === "true";
   275→      } else if (k === "passMark" || k === "smtpPort" || k === "quizDefaultCount") {
   276→        const n = Number(raw);
   277→        settingsObj[k] = Number.isNaN(n) ? 0 : n;
   278→      } else {
   279→        settingsObj[k] = raw;
   280→      }
   281→    }
   282→
   283→    setSaving(true);
   284→    try {
   285→      const res = await fetch("/api/admin/settings", {
   286→        method: "PUT",
   287→        headers: { "Content-Type": "application/json" },
   288→        body: JSON.stringify({ settings: settingsObj, category: section }),
   289→      });
   290→      const json = await res.json();
   291→      if (!res.ok) throw new Error(json.error || "Failed to save settings");
   292→      toast.success("Settings saved", {
   293→        description: `${SECTIONS.find((s) => s.key === section)?.label} section updated.`,
   294→      });
   295→      setDirty((prev) => ({ ...prev, [section]: false }));
   296→      // Merge any server-returned values
   297→      if (json.data) {
   298→        const next: FormState = { ...values };
   299→        for (const k of keys) {
   300→          if (json.data[k]) next[k] = valueToString(json.data[k].value);
   301→        }
   302→        setValues(next);
   303→      }
   304→    } catch (err) {
   305→      toast.error("Save failed", {
   306→        description: err instanceof Error ? err.message : "Unknown error",
   307→      });
   308→    } finally {
   309→      setSaving(false);
   310→    }
   311→  };
   312→
   313→  const activeMeta = SECTIONS.find((s) => s.key === activeSection)!;
   314→
   315→  return (
   316→    <div className="space-y-6">
   317→      {/* Header */}
   318→      <div>
   319→        <h1 className="text-2xl lg:text-3xl font-bold tracking-tight text-slate-900 dark:text-white">
   320→          Settings
   321→        </h1>
   322→        <p className="text-sm text-slate-500 dark:text-slate-400 mt-1">
   323→          Configure academy branding, AI, email, integrations, and general
   324→          options.
   325→        </p>
   326→      </div>
   327→
   328→      <div className="flex flex-col lg:flex-row gap-6">
   329→        {/* Sidebar */}
   330→        <Card className="lg:w-64 lg:flex-shrink-0 h-fit">
   331→          <CardContent className="p-2">
   332→            <nav className="space-y-1">
   333→              {SECTIONS.map((section) => {
   334→                const Icon = section.icon;
   335→                const isActive = activeSection === section.key;
   336→                const isDirty = dirty[section.key];
   337→                return (
   338→                  <button
   339→                    key={section.key}
   340→                    onClick={() => setActiveSection(section.key)}
   341→                    className={`w-full flex items-center gap-3 px-3 py-2.5 rounded-lg text-sm transition-all text-left ${
   342→                      isActive
   343→                        ? "bg-gradient-to-r from-sky-600 to-blue-600 text-white shadow-sm"
   344→                        : "text-slate-700 dark:text-slate-200 hover:bg-slate-100 dark:hover:bg-slate-800"
   345→                    }`}
   346→                  >
   347→                    <Icon className="h-4 w-4 shrink-0" />
   348→                    <span className="flex-1 font-medium">{section.label}</span>
   349→                    {isDirty && (
   350→                      <span
   351→                        className={`h-2 w-2 rounded-full ${
   352→                          isActive ? "bg-white" : "bg-amber-500"
   353→                        }`}
   354→                        title="Unsaved changes"
   355→                      />
   356→                    )}
   357→                  </button>
   358→                );
   359→              })}
   360→            </nav>
   361→          </CardContent>
   362→        </Card>
   363→
   364→        {/* Content area */}
   365→        <div className="flex-1 min-w-0">
   366→          {loading ? (
   367→            <Card>
   368→              <CardContent className="p-6 space-y-4">
   369→                <Skeleton className="h-7 w-48" />
   370→                <Skeleton className="h-4 w-72" />
   371→                <div className="grid grid-cols-1 sm:grid-cols-2 gap-4 pt-4">
   372→                  {Array.from({ length: 6 }).map((_, i) => (
   373→                    <Skeleton key={i} className="h-16 w-full" />
   374→                  ))}
   375→                </div>
   376→              </CardContent>
   377→            </Card>
   378→          ) : (
   379→            <Card>
   380→              <CardHeader className="border-b border-slate-200 dark:border-slate-800">
   381→                <div className="flex items-start justify-between gap-3">
   382→                  <div>
   383→                    <CardTitle className="flex items-center gap-2">
   384→                      <activeMeta.icon className="h-5 w-5 text-sky-600 dark:text-sky-400" />
   385→                      {activeMeta.label}
   386→                    </CardTitle>
   387→                    <CardDescription className="mt-1">
   388→                      {activeMeta.description}
   389→                    </CardDescription>
   390→                  </div>
   391→                  {dirty[activeSection] && (
   392→                    <Badge
   393→                      variant="outline"
   394→                      className="text-amber-700 border-amber-300 bg-amber-50 dark:bg-amber-950/40 dark:text-amber-300 dark:border-amber-800"
   395→                    >
   396→                      Unsaved changes
   397→                    </Badge>
   398→                  )}
   399→                </div>
   400→              </CardHeader>
   401→              <CardContent className="p-6">
   402→                <ScrollArea className="h-[calc(100vh-22rem)]">
   403→                  <div className="pr-2">
   404→                    {activeSection === "BRANDING" && (
   405→                      <BrandingSection
   406→                        values={values}
   407→                        updateField={updateField}
   408→                      />
   409→                    )}
   410→                    {activeSection === "AI" && (
   411→                      <AISection
   412→                        values={values}
   413→                        updateField={updateField}
   414→                        updateBool={updateBool}
   415→                      />
   416→                    )}
   417→                    {activeSection === "EMAIL" && (
   418→                      <EmailSection values={values} updateField={updateField} />
   419→                    )}
   420→                    {activeSection === "INTEGRATION" && (
   421→                      <IntegrationSection
   422→                        values={values}
   423→                        updateField={updateField}
   424→                      />
   425→                    )}
   426→                    {activeSection === "GENERAL" && (
   427→                      <GeneralSection
   428→                        values={values}
   429→                        updateField={updateField}
   430→                        updateBool={updateBool}
   431→                      />
   432→                    )}
   433→
   434→                    {/* Save button */}
   435→                    <div className="flex items-center justify-end gap-3 pt-6 mt-6 border-t border-slate-200 dark:border-slate-800">
   436→                      <Button
   437→                        variant="outline"
   438→                        onClick={fetchSettings}
   439→                        disabled={saving}
   440→                      >
   441→                        Reset
   442→                      </Button>
   443→                      <Button
   444→                        onClick={() => handleSaveSection(activeSection)}
   445→                        disabled={saving || !dirty[activeSection]}
   446→                        className="gap-2"
   447→                      >
   448→                        {saving ? (
   449→                          <Loader2 className="h-4 w-4 animate-spin" />
   450→                        ) : (
   451→                          <Save className="h-4 w-4" />
   452→                        )}
   453→                        Save Changes
   454→                      </Button>
   455→                    </div>
   456→                  </div>
   457→                </ScrollArea>
   458→              </CardContent>
   459→            </Card>
   460→          )}
   461→        </div>
   462→      </div>
   463→    </div>
   464→  );
   465→}

   497→function ColorField({
