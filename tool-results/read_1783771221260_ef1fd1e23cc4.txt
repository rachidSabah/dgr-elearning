    49→const SETTINGS_KEY = "dgr-academy-settings";

    51→function getSettings(): Record<string, string> {
    52→  if (typeof window === "undefined") return {};
    53→  const data = localStorage.getItem(SETTINGS_KEY);
    54→  return data ? JSON.parse(data) : {};
    55→}
    56→
    57→function saveSettings(settings: Record<string, string>) {
    58→  if (typeof window === "undefined") return;
    59→  localStorage.setItem(SETTINGS_KEY, JSON.stringify(settings));
    60→}

   208→function valueToString(v: unknown): string {
   209→  if (v === null || v === undefined) return "";
   210→  if (typeof v === "string") return v;
   211→  if (typeof v === "boolean") return v ? "true" : "false";
   212→  return String(v);
   213→}

   216→export default function SettingsAdminPage() {
   217→  const [activeSection, setActiveSection] = useState<SectionKey>("BRANDING");
   218→  const [values, setValues] = useState<FormState>({ ...DEFAULTS });
   219→  const [loading, setLoading] = useState(true);
   220→  const [saving, setSaving] = useState(false);
   221→  const [dirty, setDirty] = useState<Record<SectionKey, boolean>>({
   222→    BRANDING: false,
   223→    AI: false,
   224→    EMAIL: false,
   225→    INTEGRATION: false,
   226→    GENERAL: false,
   227→  });
   228→
   229→  const fetchSettings = useCallback(() => {
   230→    setLoading(true);
   231→    try {
   232→      const stored = getSettings();
   233→      const next: FormState = { ...DEFAULTS };
   234→      for (const [key, value] of Object.entries(stored)) {
   235→        next[key] = value;
   236→      }
   237→      setValues(next);
   238→      setDirty({
   239→        BRANDING: false,
   240→        AI: false,
   241→        EMAIL: false,
   242→        INTEGRATION: false,
   243→        GENERAL: false,
   244→      });
   245→    } catch (err) {
   246→      toast.error("Failed to load settings", {
   247→        description: err instanceof Error ? err.message : "Unknown error",
   248→      });
   249→    } finally {
   250→      setLoading(false);
   251→    }
   252→  }, []);
   253→
   254→  useEffect(() => {
   255→    fetchSettings();
   256→  }, [fetchSettings]);
   257→
   258→  const updateField = (key: string, val: string) => {
   259→    setValues((prev) => ({ ...prev, [key]: val }));
   260→    // Mark the relevant section dirty
   261→    for (const [sectionKey, keys] of Object.entries(SECTION_KEYS)) {
   262→      if (keys.includes(key)) {
   263→        setDirty((prev) => ({ ...prev, [sectionKey]: true }));
   264→        break;
   265→      }
   266→    }
   267→  };
   268→
   269→  const updateBool = (key: string, val: boolean) => {
   270→    updateField(key, val ? "true" : "false");
   271→  };
   272→
   273→  const handleSaveSection = (section: SectionKey) => {
   274→    const keys = SECTION_KEYS[section];
   275→    const settingsObj: Record<string, string> = {};
   276→    for (const k of keys) {
   277→      const raw = values[k] ?? "";
   278→      if (BOOL_KEYS.has(k)) {
   279→        settingsObj[k] = raw === "true" ? "true" : "false";
   280→      } else {
   281→        settingsObj[k] = raw;
   282→      }
   283→    }
   284→
   285→    setSaving(true);
   286→    try {
   287→      // Merge into existing localStorage settings and persist
   288→      const current = getSettings();
   289→      const merged = { ...current, ...settingsObj };
   290→      saveSettings(merged);
   291→      toast.success("Settings saved", {
   292→        description: `${SECTIONS.find((s) => s.key === section)?.label} section updated.`,
   293→      });
   294→      setDirty((prev) => ({ ...prev, [section]: false }));
   295→    } catch (err) {
   296→      toast.error("Save failed", {
   297→        description: err instanceof Error ? err.message : "Unknown error",
   298→      });
   299→    } finally {
   300→      setSaving(false);
   301→    }
   302→  };
   303→
   304→  const activeMeta = SECTIONS.find((s) => s.key === activeSection)!;
   305→
   306→  return (
   307→    <div className="space-y-6">
   308→      {/* Header */}
   309→      <div>
   310→        <h1 className="text-2xl lg:text-3xl font-bold tracking-tight text-slate-900 dark:text-white">
   311→          Settings
   312→        </h1>
   313→        <p className="text-sm text-slate-500 dark:text-slate-400 mt-1">
   314→          Configure academy branding, AI, email, integrations, and general
   315→          options.
   316→        </p>
   317→      </div>
   318→
   319→      <div className="flex flex-col lg:flex-row gap-6">
   320→        {/* Sidebar */}
   321→        <Card className="lg:w-64 lg:flex-shrink-0 h-fit">
   322→          <CardContent className="p-2">
   323→            <nav className="space-y-1">
   324→              {SECTIONS.map((section) => {
   325→                const Icon = section.icon;
   326→                const isActive = activeSection === section.key;
   327→                const isDirty = dirty[section.key];
   328→                return (
   329→                  <button
   330→                    key={section.key}
   331→                    onClick={() => setActiveSection(section.key)}
   332→                    className={`w-full flex items-center gap-3 px-3 py-2.5 rounded-lg text-sm transition-all text-left ${
   333→                      isActive
   334→                        ? "bg-gradient-to-r from-sky-600 to-blue-600 text-white shadow-sm"
   335→                        : "text-slate-700 dark:text-slate-200 hover:bg-slate-100 dark:hover:bg-slate-800"
   336→                    }`}
   337→                  >
   338→                    <Icon className="h-4 w-4 shrink-0" />
   339→                    <span className="flex-1 font-medium">{section.label}</span>
   340→                    {isDirty && (
   341→                      <span
   342→                        className={`h-2 w-2 rounded-full ${
   343→                          isActive ? "bg-white" : "bg-amber-500"
   344→                        }`}
   345→                        title="Unsaved changes"
   346→                      />
   347→                    )}
   348→                  </button>
   349→                );
   350→              })}
   351→            </nav>
   352→          </CardContent>
   353→        </Card>
   354→
   355→        {/* Content area */}
   356→        <div className="flex-1 min-w-0">
   357→          {loading ? (
   358→            <Card>
   359→              <CardContent className="p-6 space-y-4">
   360→                <Skeleton className="h-7 w-48" />
   361→                <Skeleton className="h-4 w-72" />
   362→                <div className="grid grid-cols-1 sm:grid-cols-2 gap-4 pt-4">
   363→                  {Array.from({ length: 6 }).map((_, i) => (
   364→                    <Skeleton key={i} className="h-16 w-full" />
   365→                  ))}
   366→                </div>
   367→              </CardContent>
   368→            </Card>
   369→          ) : (
   370→            <Card>
   371→              <CardHeader className="border-b border-slate-200 dark:border-slate-800">
   372→                <div className="flex items-start justify-between gap-3">
   373→                  <div>
   374→                    <CardTitle className="flex items-center gap-2">
   375→                      <activeMeta.icon className="h-5 w-5 text-sky-600 dark:text-sky-400" />
   376→                      {activeMeta.label}
   377→                    </CardTitle>
   378→                    <CardDescription className="mt-1">
   379→                      {activeMeta.description}
   380→                    </CardDescription>
   381→                  </div>
   382→                  {dirty[activeSection] && (
   383→                    <Badge
   384→                      variant="outline"
   385→                      className="text-amber-700 border-amber-300 bg-amber-50 dark:bg-amber-950/40 dark:text-amber-300 dark:border-amber-800"
   386→                    >
   387→                      Unsaved changes
   388→                    </Badge>
   389→                  )}
   390→                </div>
   391→              </CardHeader>
   392→              <CardContent className="p-6">
   393→                <ScrollArea className="h-[calc(100vh-22rem)]">
   394→                  <div className="pr-2">
   395→                    {activeSection === "BRANDING" && (
   396→                      <BrandingSection
   397→                        values={values}
   398→                        updateField={updateField}
   399→                      />
   400→                    )}
   401→                    {activeSection === "AI" && (
   402→                      <AISection
   403→                        values={values}
   404→                        updateField={updateField}
   405→                        updateBool={updateBool}
   406→                      />
   407→                    )}
   408→                    {activeSection === "EMAIL" && (
   409→                      <EmailSection values={values} updateField={updateField} />
   410→                    )}
   411→                    {activeSection === "INTEGRATION" && (
   412→                      <IntegrationSection
   413→                        values={values}
   414→                        updateField={updateField}
   415→                      />
   416→                    )}
   417→                    {activeSection === "GENERAL" && (
   418→                      <GeneralSection
   419→                        values={values}
   420→                        updateField={updateField}
   421→                        updateBool={updateBool}
   422→                      />
   423→                    )}
   424→
   425→                    {/* Save button */}
   426→                    <div className="flex items-center justify-end gap-3 pt-6 mt-6 border-t border-slate-200 dark:border-slate-800">
   427→                      <Button
   428→                        variant="outline"
   429→                        onClick={fetchSettings}
   430→                        disabled={saving}
   431→                      >
   432→                        Reset
   433→                      </Button>

   435→                        onClick={() => handleSaveSection(activeSection)}

   438→                      >
   439→                        {saving ? (
   440→                          <Loader2 className="h-4 w-4 animate-spin" />
   441→                        ) : (
   442→                          <Save className="h-4 w-4" />
   443→                        )}
   444→                        Save Changes
   445→                      </Button>
   446→                    </div>
   447→                  </div>
   448→                </ScrollArea>
   449→              </CardContent>
   450→            </Card>
   451→          )}
   452→        </div>
   453→      </div>
   454→    </div>
   455→  );
   456→}
